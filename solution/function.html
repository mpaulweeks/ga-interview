Exercise #2 Solution


          function divide(a, b) {
            if (b === 0) {
              return 'error';
            }
            return a / b;
          }

          var dividend = divide(3, 6);
          // dividend => 0.5

          var dividend = divide(5, 0);
          // dividend => 'error'
